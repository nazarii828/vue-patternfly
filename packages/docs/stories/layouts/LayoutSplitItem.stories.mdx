import PfSkeleton from '@vue-patternfly/core';

  components: { PfSplit, PfSplitItem },

# PfSplit `pf-split-item`

## Examples

### Basic

The dashed lines are not part of the component and are only for visual separation in the documentation. The highlighted item with green lines is the one controlled by the story.

<Canvas>
  <Story
    name="Basic"
    args={{
      filled: true,
    }}
    decorators={[
      () => ({
        components: { PfSplit, PfSplitItem },
        template: `
          <pf-split gutter class="docs-layout-outline">
            <pf-split-item>content</pf-split-item>
            <story style="border-color: var(--pf-global--palette--green-300)" />
            <pf-split-item>content</pf-split-item>
          </pf-split>
        `,
      }),
    ]}>
    {Template.bind({})}
  </Story>
</Canvas>

### Skeleton items

We can use `PfSkeleton` as the `component` prop of each `pf-split-item` to create a skeleton layout.

<Canvas>
  <Story
    name="Skeleton items"
    args={{
      filled: true,
      component: PfSkeleton,
    }}
    decorators={[
      () => ({
        components: { PfSplit, PfSplitItem },
        setup() {
          return {
            component: markRaw(PfSkeleton),
          };
        },
        template: `
          <!-- storybook wants to transform the root element to a skeleton itself, so we give it a div to play with -->
          <div>
            <pf-split gutter>
              <pf-split-item width="15%" :component="component">content</pf-split-item>
              <story style="border-color: var(--pf-global--palette--green-300)" />
              <pf-split-item width="15%" :component="component">content</pf-split-item>
            </pf-split>
          </div>
        `,
      }),
    ]}>
    {Template.bind({})}
  </Story>
</Canvas>
